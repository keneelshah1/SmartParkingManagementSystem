#include <iostream>
using namespace std;

class Time
{
    int hours,minutes,seconds;
    
    public:
    Time ();
    Time (int,int,int);
    void show_time();
    void add(Time,Time);
};

Time :: Time()
{
    hours=0;
    minutes=0;
    seconds=0;
}

Time :: Time(int h,int m,int s)
{
    hours=h;
    minutes=m;
    seconds=s;
}

void Time :: show_time()
{
    cout << hours << ":" << minutes << ":" << seconds << endl;
}

void Time :: add(Time t1, Time t2)
{
    hours=t1.hours+t2.hours;
    minutes=t1.minutes+t2.minutes;
    seconds=t1.seconds+t2.seconds;
    if(seconds > 60)
    {
        minutes=minutes+(seconds/60);
        seconds=seconds%60;
    }
    if(minutes > 60)
    {
        hours=hours+(minutes/60);
        minutes=minutes%60;
    }
}

int main()
{
    Time t1,t4;
    t1.show_time();
    Time t2 (50,50,50);
    t2.show_time();
    Time t3 (15,15,15);
    t3.show_time();
    t4.add(t2,t3);
    t4.show_time();
}
